def clean_price(price):
    if isinstance(price, str):
        return float(re.sub(r'[,]', '', price))  # 
    return price

df['discounted_price'] = df['discounted_price'].apply(clean_price)
df['actual_price'] = df['actual_price'].apply(clean_price)
df['discount_percentage'] = df['discount_percentage'].str.replace('%', '').astype(float) / 100  # 
df['rating'] = pd.to_numeric(df['rating'].replace('|', np.nan), errors='coerce')  # '|'NaN
df['rating_count'] = pd.to_numeric(df['rating_count'].str.replace(',', ''), errors='coerce')  # 

# 
df['main_category'] = df['category'].str.split('|').str[0]  # 

# 
df = df.dropna(subset=['rating', 'rating_count'])  # 
df.fillna({'review_content': ''}, inplace=True)  # 

print("\n:", df.shape)
print("\n:\n", df[['discounted_price', 'actual_price', 'rating', 'rating_count']].describe())
